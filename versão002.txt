// Utilitário para retentativas com backoff exponencial
async function fetchWithRetry(url, options, maxRetries = 5) {
  let delay = 1000;
  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch(url, options);
      if (response.ok) return await response.json();
      if (response.status !== 429 && response.status < 500) break; 
    } catch (e) {
      if (i === maxRetries - 1) throw e;
    }
    await new Promise(res => setTimeout(res, delay));
    delay *= 2;
  }
  return null;
}

// Motor de Tradução (Preserva a notação de roteiro)
async function translateToEnglish(text) {
  if (!text || text.trim().length < 3) return text;
  const payload = {
    contents: [{ parts: [{ text: `Translate to technical cinema English, strictly preserving script notations like INT./EXT. and DAY/NIGHT. Maintain all descriptive details: ${text}` }] }],
    systemInstruction: { parts: [{ text: "You are an expert cinema translator. Return ONLY the translation. Maintain a technical and descriptive tone." }] }
  };
  const result = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
  return result?.candidates?.[0]?.content?.parts?.[0]?.text?.trim() || text;
}

// Motor de Refinamento Criativo (Botão Mágica)
async function enhanceSceneText(text) {
  if (!text || text.trim().length < 3) return text;
  const payload = {
    contents: [{ parts: [{ text: `Improve this scene description for a visual script: ${text}` }] }],
    systemInstruction: { parts: [{ text: "You are a professional screenwriter. Improve the description visually in Portuguese. ALWAYS start with a screenplay slugline (e.g., INT. LOCAL - DIA). Follow with 2-3 sentences using sensory language." }] }
  };
  const result = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
  return result?.candidates?.[0]?.content?.parts?.[0]?.text?.trim() || text;
}

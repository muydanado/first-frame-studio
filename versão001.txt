// A chave é injetada automaticamente pelo ambiente de execução.
const apiKey = ""; 

// Utilitário para retentativas com backoff exponencial para lidar com limites de taxa
async function fetchWithRetry(url, options, maxRetries = 5) {
  let delay = 1000;
  for (let i = 0; i < maxRetries; i++) {
    try {
      const response = await fetch(url, options);
      if (response.ok) return await response.json();
      if (response.status !== 429 && response.status < 500) break; 
    } catch (e) {
      if (i === maxRetries - 1) throw e;
    }
    await new Promise(res => setTimeout(res, delay));
    delay *= 2;
  }
  return null;
}

// Motor de Tradução (Português -> Inglês Técnico)
async function translateToEnglish(text) {
  if (!text || text.trim().length < 3) return text;
  const payload = {
    contents: [{ parts: [{ text: `Translate to technical cinema English, strictly preserving script notations like INT./EXT. and DAY/NIGHT. Maintain all descriptive details: ${text}` }] }],
    systemInstruction: { parts: [{ text: "You are an expert cinema translator. Return ONLY the translation. Maintain a technical and descriptive tone." }] }
  };
  const result = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
  return result?.candidates?.[0]?.content?.parts?.[0]?.text?.trim() || text;
}

// Motor de Refinamento Criativo (Botão Mágica)
async function enhanceSceneText(text) {
  if (!text || text.trim().length < 3) return text;
  const payload = {
    contents: [{ parts: [{ text: `Improve this scene description for a visual script: ${text}` }] }],
    systemInstruction: { parts: [{ text: "You are a professional screenwriter. Improve the description to be visually rich in Portuguese. ALWAYS start with a screenplay slugline (e.g., INT. LOCAL - DIA). Follow with 2-3 sentences using sensory language." }] }
  };
  const result = await fetchWithRetry(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(payload)
  });
  return result?.candidates?.[0]?.content?.parts?.[0]?.text?.trim() || text;
}

                  <select className="w-full bg-neutral-900 border border-neutral-800 rounded-lg p-3 text-xs focus:border-red-600 outline-none cursor-pointer shadow-md appearance-none text-neutral-300" value={config.camera} onChange={(e) => setConfig({...config, camera: e.target.value})}>
                    {CAMERAS.map(c => <option key={`cam-${c.id}`} value={c.id}>{c.label}</option>)}
                  </select>
                </div>
                <div className="grid grid-cols-2 gap-4">
                  <div className="space-y-2">
                    <label className="text-[10px] font-bold text-neutral-500 uppercase flex items-center gap-2 tracking-widest">
                      <Focus className="w-3.5 h-3.5 text-red-500" /> Lente
                    </label>
                    <select className="w-full bg-neutral-900 border border-neutral-800 rounded-lg p-3 text-xs focus:border-red-600 outline-none cursor-pointer shadow-md appearance-none text-neutral-300" value={config.lens} onChange={(e) => handleLensChange(e.target.value)}>

                    </select>
                  </div>
                  <div className="space-y-2">
                    <label className="text-[10px] font-bold text-neutral-500 uppercase flex items-center gap-2 tracking-widest">
                      <Target className="w-3.5 h-3.5 text-red-500" /> Focal

                      {FOCAL_LENGTHS.map(f => <option key={`foc-${f}`} value={f}>{f}</option>)}
                    </select>

                </div>
              </div>

              <div className="space-y-6 pt-4 border-t border-neutral-800/30">
                <div className="space-y-2">

                  </label>

                    {Object.keys(LOOKS).map(group => (
                      <optgroup key={`group-${group}`} label={group} className="bg-neutral-900 text-neutral-500 italic">
                        {LOOKS[group].map(l => <option key={`look-${l.id}`} value={l.id}>{l.label}</option>)}
                      </optgroup>
                    ))}
                  </select>
                </div>

                <div className="space-y-3">
                  <label className="text-[10px] font-bold text-neutral-500 uppercase tracking-widest flex items-center gap-2">
                    <Maximize className="w-3.5 h-3.5 text-red-500" /> Aspect Ratio
                  </label>
                  <div className="grid grid-cols-5 gap-2">
                    {ASPECT_RATIOS.map(r => (
                      <button key={`ratio-${r.id}`} onClick={() => setConfig({...config, ratio: r.id})} className={`py-2.5 text-[9px] font-black rounded-lg border transition-all duration-300 ${config.ratio === r.id ? 'bg-red-600 border-red-600 text-white shadow-lg shadow-red-900/40' : 'bg-neutral-900 border-neutral-800 text-neutral-500 hover:border-neutral-600 hover:text-neutral-300'}`}>
                        {r.label.split(' ')[0]}
                      </button>
                    ))}
                  </div>
                </div>
              </div>
            </div>
          </div>
        </aside>

        {/* ÁREA PRINCIPAL */}
        <main className="flex-1 bg-neutral-950 p-12 overflow-y-auto relative flex flex-col items-center h-full custom-scrollbar">
          <div className="w-full max-w-5xl space-y-12 pb-20">
            
            <div className="flex items-center justify-between border-l-4 border-red-600 pl-8 py-3 bg-neutral-900/20 rounded-r-2xl pr-6">
              <div>
                <h2 className="text-xl font-black uppercase italic tracking-tighter text-white/40 mb-1">Unidade de Análise de Cena</h2>
                <p className="text-[10px] text-neutral-500 font-mono tracking-[0.3em] uppercase">Metadados Técnicos em Tempo Real</p>
              </div>
              <div className="flex items-center gap-10">
                <div className="text-right">
                  <p className="text-[9px] font-bold text-neutral-600 uppercase tracking-widest mb-1.5">Status</p>
                  <div className="flex items-center gap-2.5 justify-end">
                     <span className="w-2 h-2 rounded-full bg-red-600 animate-pulse shadow-sm shadow-red-500"></span>
                     <p className="text-xs font-mono text-red-500 font-black uppercase tracking-widest">LIVE SYNC</p>
                  </div>
                </div>
              </div>
            </div>

            <div className="space-y-5">
              <h3 className="text-[11px] font-black text-neutral-600 uppercase tracking-[0.3em] px-1">Hierarquia Modular (Nano Banana v1)</h3>
              <div className="grid gap-3">
                {Object.entries(promptBlocks).map(([key, value]) => (
                  <div key={`block-${key}`} className="flex bg-neutral-900/30 rounded-2xl border border-neutral-800/60 group hover:border-red-900/40 transition-all duration-300 overflow-hidden shadow-sm">

                      {value}
                    </div>
                  </div>
                ))}
              </div>
            </div>

              <div className="flex items-center justify-between border-l-4 border-red-600 pl-8 py-3 bg-neutral-900/20 rounded-r-2xl pr-6">
                <div>
                  <h3 className="text-xl font-black uppercase italic tracking-tighter text-white/40 leading-none mb-1">Output Estruturado</h3>
                  <div className="flex items-center gap-4 mt-2">
                    <button

                      className={`flex items-center gap-2 text-[10px] font-bold uppercase tracking-widest transition-all ${outputMode === 'image' ? 'text-red-500' : 'text-neutral-600 hover:text-neutral-400'}`}
                    >
                      <Image className="w-3.5 h-3.5" /> Frame

                    <div className="w-px h-3 bg-neutral-800"></div>
                    <button 
                      onClick={() => setOutputMode('video')}
                      className={`flex items-center gap-2 text-[10px] font-bold uppercase tracking-widest transition-all ${outputMode === 'video' ? 'text-red-500' : 'text-neutral-600 hover:text-neutral-400'}`}
                    >
                      <Play className="w-3.5 h-3.5" /> Motion (Video)
                    </button>
                  </div>
                </div>
                <button onClick={copyToClipboard} disabled={!translatedAction || isTranslating || isAnalyzing} className={`px-14 py-6 rounded-full font-black text-sm flex items-center gap-4 transition-all duration-500 shadow-2xl uppercase tracking-[0.2em] active:scale-95 ${!translatedAction || isTranslating || isAnalyzing ? 'bg-neutral-800 text-neutral-600 opacity-40' : copied ? 'bg-emerald-600 text-white shadow-emerald-900/20' : 'bg-red-600 text-white hover:scale-105 shadow-red-900/40'}`}>
                  {copied ? <Check className="w-6 h-6" /> : <Copy className="w-6 h-6" />}
                  {copied ? 'Copiado!' : `Copiar ${outputMode === 'video' ? 'Video' : 'Frame'}`}
                </button>
              </div>
              
              <div className="relative group overflow-hidden rounded-[2.5rem]">
                <div className={`w-full bg-black font-mono text-sm p-10 pt-14 rounded-[2.5rem] shadow-[inset_0_0_80px_rgba(0,0,0,0.8)] border border-neutral-800/80 leading-relaxed min-h-[160px] relative transition-all duration-700 group-hover:border-red-900/40 ${outputMode === 'video' ? 'text-amber-50 selection:bg-amber-600' : 'text-red-50 selection:bg-red-600'}`}>
                  {isTranslating || isEnhancing || isAnalyzing ? (
                    <div className="flex flex-col items-center justify-center gap-6 text-neutral-800 italic h-full py-16">

                      <span className="text-sm font-black uppercase tracking-[0.4em] animate-pulse">Processando IA...</span>

                  ) : (
                    <span className="relative z-10 block leading-relaxed animate-fade-in">
                      {outputMode === 'video' ? finalVideoPrompt : finalPrompt}
                    </span>
                  )}

                    <Dna className="w-64 h-64 text-white rotate-12" />
                  </div>
                  <div className="absolute inset-0 pointer-events-none overflow-hidden opacity-10">
                    <div className="w-full h-full bg-[linear-gradient(rgba(18,16,16,0)_50%,rgba(0,0,0,0.25)_50%),linear-gradient(90deg,rgba(255,0,0,0.06),rgba(0,255,0,0.02),rgba(0,0,255,0.06))] bg-[length:100%_4px,3px_100%]"></div>
                  </div>
                </div>
              </div>
            </div>

            <div className="bg-neutral-900/30 border border-neutral-800/50 p-8 rounded-3xl flex gap-8 italic shadow-2xl backdrop-blur-sm">
               <div className="p-4 bg-black/40 rounded-2xl h-fit border border-neutral-800 shadow-inner">
                 <Info className="w-7 h-7 text-neutral-600" />
               </div>
               <div className="space-y-3">
                 <p className="text-xs text-neutral-400 font-black uppercase tracking-widest leading-none">Dica de Produção Industrial</p>

                   {outputMode === 'image'

                 </p>

            </footer>
          </div>
        </main>
      </div>

      <style dangerouslySetInnerHTML={{ __html: `

      `}} />
